`Some tests invloving ints and parameters`
Object subclass: IntTests 
    category: examples
    vars: v (
        init (
            v = 7;
            self localTest: "Param";
            v = v * 3 - -7 * 2;
            v println;
            (self add1: 2) println
        )

        localTest: param
        |x y z a b c d e f g h i j k l m n o p q r s|
        (
            param println;
            s = "Local s";
            param = s;
            s println;
            v println
        )

        add1: x (
            x = x + 1;
            ^x
        )
)
