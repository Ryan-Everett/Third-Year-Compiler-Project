Object subclass: LoopTest
    category: examples
    vars: i(
        init (
            i = "INSTANCE";
            self makeLoop: "LOCAL";
            examples/LoopTest2 new;
            i println;
        )
        makeLoop: i  |n k| (
            i println;
            i = 0;
            k = 20;

        )
)
